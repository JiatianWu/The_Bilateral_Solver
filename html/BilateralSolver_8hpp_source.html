148  void FastBilateralSolverFilterImpl::init(cv::Mat& reference_bgr, double sigma_spatial, double sigma_luma, double sigma_chroma)
149  {
150 
151  cv::Mat reference_yuv;
152  cv::cvtColor(reference_bgr, reference_yuv, COLOR_BGR2YCrCb);
153 
154  std::chrono::steady_clock::time_point begin_grid_construction = std::chrono::steady_clock::now();
155 
156  cols = reference_yuv.cols;
157  rows = reference_yuv.rows;
158  npixels = cols*rows;
159  std::int64_t hash_vec[5];
160  for (int i = 0; i < 5; ++i)
161  hash_vec[i] = static_cast<std::int64_t>(std::pow(255, i));
162 
163  std::unordered_map<std::int64_t /* hash */, int /* vert id */> hashed_coords;
164  hashed_coords.reserve(cols*rows);
165 
166  const unsigned char* pref = (const unsigned char*)reference_yuv.data;
167  int vert_idx = 0;
168  int pix_idx = 0;
169 
170 
171  // construct Splat(Slice) matrices
172  splat_idx.resize(npixels);
173  for (int y = 0; y < rows; ++y)
174  {
175  for (int x = 0; x < cols; ++x)
176  {
177  std::int64_t coord[5];
178  coord[0] = int(x / sigma_spatial);
179  coord[1] = int(y / sigma_spatial);
180  coord[2] = int(pref[0] / sigma_luma);
181  coord[3] = int(pref[1] / sigma_chroma);
182  coord[4] = int(pref[2] / sigma_chroma);
183 
184  // convert the coordinate to a hash value
185  std::int64_t hash_coord = 0;
186  for (int i = 0; i < 5; ++i)
187  hash_coord += coord[i] * hash_vec[i];
188 
189  // pixels whom are alike will have the same hash value.
190  // We only want to keep a unique list of hash values, therefore make sure we only insert
191  // unique hash values.
192  std::unordered_map<int64_t,int>::iterator it = hashed_coords.find(hash_coord);
193  if (it == hashed_coords.end())
194  {
195  hashed_coords.insert(std::pair<std::int64_t, int>(hash_coord, vert_idx));
196  splat_idx[pix_idx] = vert_idx;
197  ++vert_idx;
198  }
199  else
200  {
201  splat_idx[pix_idx] = it->second;
202  }
203 
204  pref += 3; // skip 3 bytes (y u v)
205  ++pix_idx;
206  }
207  }
208  nvertices = hashed_coords.size();
209 
210 
211  // construct Blur matrices
212  std::chrono::steady_clock::time_point begin_blur_construction = std::chrono::steady_clock::now();
213  Eigen::VectorXf ones_nvertices = Eigen::VectorXf::Ones(nvertices);
214  Eigen::VectorXf ones_npixels = Eigen::VectorXf::Ones(npixels);
215  blurs_test = ones_nvertices.asDiagonal();
216  blurs_test *= 10;
217  for(int offset = -1; offset <= 1;++offset)
218  {
219  if(offset == 0) continue;
220  for (int i = 0; i < 5; ++i)
221  {
222  Eigen::SparseMatrix<float, Eigen::ColMajor> blur_temp(hashed_coords.size(), hashed_coords.size());
223  blur_temp.reserve(Eigen::VectorXi::Constant(nvertices,6));
224  std::int64_t offset_hash_coord = offset * hash_vec[i];
225  for (std::unordered_map<int64_t,int>::iterator it = hashed_coords.begin(); it != hashed_coords.end(); ++it)
226  {
227  std::int64_t neighb_coord = it->first + offset_hash_coord;
228  std::unordered_map<int64_t,int>::iterator it_neighb = hashed_coords.find(neighb_coord);
229  if (it_neighb != hashed_coords.end())
230  {
231  blur_temp.insert(it->second,it_neighb->second) = 1.0f;
232  blur_idx.push_back(std::pair<int,int>(it->second, it_neighb->second));
233  }
234  }
235  blurs_test += blur_temp;
236  }
237  }
238  blurs_test.finalize();
239 
240 
241  //bistochastize
242  int maxiter = 10;
243  n = ones_nvertices;
244  m = Eigen::VectorXf::Zero(nvertices);
245  for (int i = 0; i < splat_idx.size(); i++) {
246  m(splat_idx[i]) += 1.0f;
247  }
248 
249  Eigen::VectorXf bluredn(nvertices);
250 
251  for (int i = 0; i < maxiter; i++) {
252  Blur(n,bluredn);
253  n = ((n.array()*m.array()).array()/bluredn.array()).array().sqrt();
254  }
255  Blur(n,bluredn);
256 
257  m = n.array() * (bluredn).array();
258  Dm = m.asDiagonal();
259  Dn = n.asDiagonal();
260 
261 
262  std::cout << "Splat"<< splat_idx.size() << '\n';
263  std::cout << "Blur"<< blurs_test.nonZeros() << '\n';
264  std::cout << "Dn"<< Dn.nonZeros() << '\n';
265  std::cout << "Dm"<< Dm.nonZeros() << '\n';
266  }

297  void FastBilateralSolverFilterImpl::solve(cv::Mat& target,
298  cv::Mat& confidence,
299  cv::Mat& output)
300  {
301 
302  Eigen::VectorXf x(npixels);
303  Eigen::VectorXf w(npixels);
304 
305  const uchar *pft = reinterpret_cast<const uchar*>(target.data);
306  for (int i = 0; i < npixels; i++)
307  {
308  x(i) = float(pft[i])/255.0f;
309  }
310  const uchar *pfc = reinterpret_cast<const uchar*>(confidence.data);
311  for (int i = 0; i < npixels; i++)
312  {
313  w(i) = float(pfc[i])/255.0f;
314  }
315 
316  Eigen::SparseMatrix<float, Eigen::ColMajor> M(nvertices,nvertices);
317  Eigen::SparseMatrix<float, Eigen::ColMajor> A_data(nvertices,nvertices);
318  Eigen::SparseMatrix<float, Eigen::ColMajor> A(nvertices,nvertices);
319  Eigen::VectorXf b(nvertices);
320  Eigen::VectorXf y(nvertices);
321  Eigen::VectorXf w_splat(nvertices);
322  Eigen::VectorXf xw(x.size());
323 
324 
325 
326  //construct A
327  Splat(w,w_splat);
328  A_data = (w_splat).asDiagonal();
329  A = bs_param.lam * (Dm - Dn * (blurs_test*Dn)) + A_data ;
330 
331  //construct b
332  b.setZero();
333  for (int i = 0; i < splat_idx.size(); i++) {
334  b(splat_idx[i]) += x(i) * w(i);
335  }
336 
337  // solve Ay = b
338  Eigen::ConjugateGradient<Eigen::SparseMatrix<float>, Eigen::Lower|Eigen::Upper> cg;
339  cg.compute(A);
340  cg.setMaxIterations(bs_param.cg_maxiter);
341  cg.setTolerance(bs_param.cg_tol);
342  y = cg.solve(b);
343  // std::cout << "#iterations: " << cg.iterations() << std::endl;
344  // std::cout << "estimated error: " << cg.error() << std::endl;
345 
346  //slice
347  uchar *pftar = (uchar*)(output.data);
348  for (int i = 0; i < splat_idx.size(); i++)
349  {
350  pftar[i] = y(splat_idx[i]) * 255;
351  }
352 
353 
354  }
